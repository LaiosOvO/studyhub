    ```python
    class GitManager:
        def __init__(self, workspace: Path): ...
        def init_repo(self) -> None: ...
        def create_branch(self, branch_name: str) -> None: ...
        def commit(self, message: str) -> str:
            """Stage all changes and commit. Return SHA hex. Immutable -- creates new commit object."""
        def get_current_sha(self) -> str: ...
        def reset_to_previous(self) -> None:
            """Reset to parent commit (discard failed iteration). Uses git reset --hard HEAD~1."""
        def is_clean(self) -> bool:
            """Check for uncommitted changes."""
        def stash_and_clean(self) -> None:
            """Safety net: stash any uncommitted changes before iteration."""
        def get_log(self, max_count: int = 50) -> list[dict]:
            """Return recent commit log as list of {sha, message, timestamp}."""
    ```

    **`backend/app/services/experiment/docker_runner.py`**: Class `DockerRunner` managing Docker container lifecycle with GPU passthrough. Reference: 08-RESEARCH.md Pattern 3.

    ```python
    class DockerRunner:
        CONTAINER_LABEL = "com.studyhub.experiment"

        def __init__(self, workspace: Path, gpu_device: int = 0, image: str = "studyhub/experiment-runner:latest"): ...

        async def preflight_check(self) -> dict:
            """Validate Docker daemon, GPU access, image available. Returns {docker_ok, gpu_ok, image_ok, errors}. Run this before starting experiments."""

        async def build_image(self, dockerfile_path: Path | None = None) -> str:
            """Build experiment runner image if not exists. Returns image tag. Uses asyncio.to_thread for blocking docker-py calls."""

        async def install_requirements(self, requirements_path: Path) -> bool:
            """Run pip install -r inside container to add project-specific deps. Returns success."""

        async def run_training(self, command: str = "python train.py", timeout_seconds: int = 3600) -> tuple[str, int]:
            """Run training in Docker container. Returns (stdout_log, exit_code).
            Container config:
            - volumes: workspace bind-mounted at /workspace (rw)
            - device_requests: GPU with device_ids=[str(gpu_device)]
            - network_mode: 'none' (sandboxed -- EXPR-08)
            - mem_limit: '32g'
            - shm_size: '8g' (PyTorch DataLoader requirement)
            - working_dir: '/workspace'
            - auto_remove: False (need to capture logs first)
            - labels: {CONTAINER_LABEL: run_id}
            After capturing logs, remove container explicitly.
            Uses asyncio.to_thread for all blocking docker-py calls.
            """

        async def stop_container(self) -> None:
            """Stop running container (for pause/cancel)."""

        async def cleanup_orphans(self) -> int:
            """Find and remove containers with studyhub label. Returns count removed. Run on app startup."""

        def _get_client(self) -> docker.DockerClient:
            """Get docker client. Lazy initialization."""
    ```

    ```python
    EXPERIMENTS_BASE = Path.home() / ".studyhub" / "experiments"

    async def setup_workspace(
        plan: ExperimentPlan,
        run_id: str,
        gpu_device: int = 0,
    ) -> Path:
        """Set up isolated experiment workspace.
        1. Create directory: ~/.studyhub/experiments/{plan_id}/{run_id}/
        2. Write code_skeleton as train.py
        3. Write plan metadata as plan.json
        4. Write requirements.txt if datasets need downloading
        5. Initialize git repo with GitManager
        6. Create branch: experiment/{run_id}
        7. Initial commit: "setup: experiment workspace from plan {plan.title}"
        Returns workspace path.
        """

    async def setup_baseline(
        workspace: Path,
        runner: DockerRunner,
        git: GitManager,
    ) -> dict | None:
        """Run baseline training and record results (EXPR-03).
        1. Run training via DockerRunner (this runs code_skeleton as-is)
        2. Parse output metrics
        3. If metrics found: commit as "baseline: {metric_name}={metric_value}", return metrics dict
        4. If no metrics (crash): return None
        """

    async def teardown_workspace(workspace: Path) -> None:
        """Clean up workspace. Archive results.tsv and .git to SeaweedFS if available, then remove directory."""

    async def download_datasets(
        workspace: Path,
        datasets: list[dict],
    ) -> list[str]:
        """Download datasets to workspace/data/. Returns list of downloaded paths. Non-fatal: logs warning on failure, continues."""
    ```
